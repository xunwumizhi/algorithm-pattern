### [208. 实现 Trie (前缀树)](https://leetcode.cn/problems/implement-trie-prefix-tree/)

```go
/**
 * Your Trie object will be instantiated and called as such:
 * obj := Constructor();
 * obj.Insert(word);
 * param_2 := obj.Search(word);
 * param_3 := obj.StartsWith(prefix);
 */
 
type Trie struct {
	childs [26]*Trie
	isEnd  bool
}
func Constructor() Trie {
	return Trie{}
}

func (t *Trie) Insert(word string) {
	root := t
	for i := 0; i < len(word); i++ {
		index := word[i] - 'a'
		if root.childs[index] == nil {
			root.childs[index] = &Trie{}
		}
		root = root.childs[index]
	}
	root.isEnd = true
}

func (t *Trie) searchPrefix(prefix string) *Trie {
	root := t
	for i := 0; i < len(prefix); i++ {
		index := prefix[i] - 'a'
		if root.childs[index] == nil {
			return nil
		}
		root = root.childs[index]
	}
	return root
}

func (t *Trie) Search(word string) bool {
	node := t.searchPrefix(word)
	return node != nil && node.isEnd
}
func (t *Trie) StartsWith(prefix string) bool {
	return t.searchPrefix(prefix) != nil
}
```

### [648. 单词替换](https://leetcode.cn/problems/replace-words/)

```go
type Trie struct {
	childs [26]*Trie
	isEnd  bool
}

func (t *Trie) Insert(word string) {
	root := t
	for i := 0; i < len(word); i++ {
		index := word[i] - 'a'
		if root.childs[index] == nil {
			root.childs[index] = &Trie{}
		}
		root = root.childs[index]
	}
	root.isEnd = true
}

func (t *Trie) getPrefix(word string) string {
	root := t
	for i := 0; i < len(word); i++ {
		index := word[i] - 'a'
		if root.childs[index] == nil {
			return word // 未匹配
		}
        if root.childs[index].isEnd {
            return word[:i+1] // 匹配到短前缀
        }
		root = root.childs[index]
	}
    // 完全被包含，则前缀就是word，前缀的前缀
	return word
}

func replaceWords(dictionary []string, sentence string) string {
    trie := &Trie{}
    for _, word := range dictionary {
        trie.Insert(word)
    }

    words := strings.Split(sentence, " ")
    for i, word := range words {
        res := trie.getPrefix(word)
        words[i] = res
    }
    return strings.Join(words, " ")
}
```

### [677. 键值映射](https://leetcode.cn/problems/map-sum-pairs/)

```go
type MapSum struct {
    root *Trie
    wordMap map[string]int
}
type Trie struct {
    childs [26]*Trie
    val int
}
func Constructor() MapSum {
    return MapSum{
        root: &Trie{},
        wordMap: map[string]int{},
    }
}

func (this *MapSum) Insert(key string, val int)  {
    // 保证完整匹配的节点值为value，路径上的前缀节点累加增量
    // 不能直接替换所有前缀的值，因为可以能累积了【前缀key】的值
    delta := val - this.wordMap[key] // 保证完整匹配的节点值为value
    this.wordMap[key] = val

    root := this.root
    for i := 0; i < len(key); i++ {
        index := key[i] - 'a'
        if root.childs[index] == nil {
            root.childs[index] = &Trie{}
        }
        node := root.childs[index]
        node.val += delta // 路径上的前缀加上对应增量
        root = node
    }
}

func (this *MapSum) Sum(prefix string) int {
    root := this.root
    for i := 0; i < len(prefix); i++ {
        index := prefix[i] - 'a'
        if root.childs[index] == nil {
            return 0
        }
        root = root.childs[index]
    }
    return root.val
}
```


### [211. 添加与搜索单词](https://leetcode.cn/problems/design-add-and-search-words-data-structure/)

```go
type WordDictionary struct {
    childs [26]*WordDictionary
    isEnd bool
}
func Constructor() WordDictionary {
    return WordDictionary{}
}
func (this *WordDictionary) AddWord(word string)  {
    root := this
    for i := 0; i < len(word); i++ {
        index := word[i] - 'a'
        if root.childs[index] == nil {
            root.childs[index] = &WordDictionary{}
        }
        root = root.childs[index]
    }
    root.isEnd = true
}

// dfs root为上一级的父节点
func (this *WordDictionary) dfs(word string, bIndex int, root *WordDictionary) bool {
    if bIndex == len(word) {
        return root.isEnd
    }

    b := word[bIndex]
    if b != '.' {
        node := root.childs[b-'a']
        if node != nil && this.dfs(word, bIndex+1, node) {
            return true
        }
    } else {
        // 当前通过，遍历下一层字符的父节点
        for i := range root.childs {  
            node := root.childs[i]
            if node != nil && this.dfs(word, bIndex+1, node) {
                return true
            }
        }
    }
    return false
}

func (this *WordDictionary) Search(word string) bool {
    return this.dfs(word, 0, this)
}
```
